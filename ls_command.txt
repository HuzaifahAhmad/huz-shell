    DIR *directory;
    struct dirent *entry;

    directory = opendir(".");

    if (directory == NULL) 
    {
        printf("Error opening directory\n");
        return -1;
    }

    while ((entry = readdir(directory)) != NULL)
    {
        printf("%s\n", entry->d_name);
    }

    if (closedir(directory) == -1)
    {
        printf("Error closing directory\n");
        return -1;
    }
